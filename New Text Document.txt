python -m compileall .''

make internal folder

@echo off
setlocal
title Excel Comparison Tool

:: Check if Python is installed
python --version >nul 2>&1
if %errorlevel% neq 0 (
    echo [!] Python is not installed or not in PATH.
    echo Please install Python to use this tool.
    pause
    exit
)

:: Move into the internal folder and run the compiled script
cd /d "%~dp0internal"
python main.pyc

if %errorlevel% neq 0 (
    echo.
    echo [!] Something went wrong.
    echo Check the 'logs/app_error.log' for details.
    pause
)
endlocal
